def parse(content)
  measure_time do
    @renderer.cache[@filename] ||= Liquid::Template.parse(content, :line_numbers => true)
  end
  @template = @renderer.cache[@filename]

  self
end

def render(*args)
  reset_template_assigns

  measure_time do
    measure_bytes do
      measure_counts do
        @template.render(*args)
      end
    end
  end
