def initialize(theme_name, opts)
  @name = theme_name.to_s.tr(" ", "_").squeeze("_")
  @path = Pathname.new(File.expand_path(name, Dir.pwd))
  @code_of_conduct = !!opts["code_of_conduct"]
end
